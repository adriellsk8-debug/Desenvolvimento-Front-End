  <script>

    function calcularIMC(){

      const peso = parseFloat(document.getElementById("peso").value);

      const altura = parseFloat(document.getElementById("altura").value);



      if(!peso || !altura){

        document.getElementById("resultadoIMC").innerText = "Preencha os campos corretamente.";

        return;

      }



      const imc = peso / (altura * altura);

      let status = "";



      if(imc < 18.5) status = "Abaixo do peso";

      else if(imc < 25) status = "Peso ideal";

      else if(imc < 30) status = "Sobrepeso";

      else status = "Obesidade";



      document.getElementById("resultadoIMC").innerText =

        `Seu IMC é ${imc.toFixed(2)} — ${status}`;

    }



    function validar(){

      const nome = document.getElementById("nome").value;

      const email = document.getElementById("email").value;



      if(nome === "" || email === ""){

        document.getElementById("formMsg").innerText =

          "Preencha nome e e-mail!";

        return false;

      }



      document.getElementById("formMsg").innerText =

        "Formulário enviado com sucesso!";



      return false; // evita recarregar a página

    }

  </script>
